USERTABLE

--Create Table
  CREATE TABLE usertable 
   (	USER_ID NUMBER(38,0) NOT NULL ENABLE, 
	EMAIL NVARCHAR2(50) NOT NULL ENABLE, 
	CREATE_DATE DATE DEFAULT sysdate NOT NULL ENABLE, 
	LAST_NAME NVARCHAR2(20) NOT NULL ENABLE, 
	FIRST_NAME NVARCHAR2(20) NOT NULL ENABLE, 
	CITY NVARCHAR2(20), 
	STATE NVARCHAR2(20), 
	PHONE NVARCHAR2(20), 
	DOB DATE NOT NULL ENABLE, 
	GENDER VARCHAR2(1 BYTE) NOT NULL ENABLE, 
	INTERESTED_IN VARCHAR2(1 BYTE) NOT NULL ENABLE, 
	PASSWORD NVARCHAR2(50) NOT NULL ENABLE, 
	DP_IMAGE BLOB, 
	DESCRIPTION NVARCHAR2(250), 
	 CONSTRAINT usertable_pk PRIMARY KEY (USER_ID)) ;
	 
 --Create sequence for User ID
CREATE SEQUENCE USER_SEQ_EMETAD START WITH 1;
-- Creating Trigger and enabling it for User ID sequence
  CREATE OR REPLACE TRIGGER USER_BIR_EMETAD 
BEFORE INSERT ON usertable 
FOR EACH ROW
BEGIN
  SELECT USER_SEQ_EMETAD.NEXTVAL
  INTO   :new.USER_ID
  FROM   dual;
END;

ALTER TRIGGER USER_BIR_EMETAD ENABLE;
 
--Creating Trigger and enabling it for automated sysdate
--col value for create_date for new row

  CREATE OR REPLACE TRIGGER DATE_TRIGGER_EMETAD 
  BEFORE INSERT OR UPDATE ON usertable
  FOR EACH ROW
BEGIN
  :new.create_date := sysdate;
END;

ALTER TRIGGER DATE_TRIGGER_EMETAD ENABLE;
